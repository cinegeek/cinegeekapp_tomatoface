//ユーザー情報
CREATE TABLE user_info (
  id BIGINT,
  name char(255),
  gender char(255),
  pic char(255),
  movies char(255),
  PRIMARY KEY (id)
)

//映画情報
CREATE TABLE movie_info (
  id BIGINT,
  name char(255),
  pic char(255),
  japanese char(255),
  coment char(255),
  PRIMARY KEY (id)
)
